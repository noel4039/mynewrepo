5.Diamond pattern
#include <stdio.h>

int main() {
    int i = 1, j, n;

    printf("Enter the number of rows: ");
    scanf("%d", &n);

    while (i <= n) {
        // Print spaces
        j = i;
        while (j < n) {
            printf(" ");
            j++;
        }
        j = 1;
        while (j <= (2 * i - 1)) {
            printf("*");
            j++;
        }
        printf("\n");
        i++;
    }

    i = n - 1;
    while (i >= 1) {
        j = n;
        while (j > i) {
            printf(" ");
            j--;
        }
        // Print stars
        j = 1;
        while (j <= (2 * i - 1)) {
            printf("*");
            j++;
        }
        printf("\n");
        i--;
    }

    return 0;
}
